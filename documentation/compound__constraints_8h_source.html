786 class UnsatisfiableConstraintInfo {
787  public:
788  UnsatisfiableConstraintInfo(const vpsc::Constraint* c);
789 
791  unsigned leftVarIndex;
793  unsigned rightVarIndex;
795  double separation;
797  bool equality;
799  cola::CompoundConstraint *cc;
800 
801  std::string toString(void) const
802  {
803  std::stringstream stream;
804  stream << "Unsatisfiable constraint: var(" << leftVarIndex << ") ";
805  if (separation < 0)
806  {
807  stream << "- " << -separation;
808  }
809  else
810  {
811  stream << "+ " << separation;
812  }
813  stream << " " << ((equality) ? "== " : "<= ");
814  stream << "var(" << rightVarIndex << ")";
815  if (cc)
816  {
817  stream << "\n From " << cc->toString();
818  }
819 
820  return stream.str();
821  }
