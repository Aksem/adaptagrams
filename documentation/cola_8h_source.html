205 class TestConvergence {
206 public:
207  double old_stress;
208  TestConvergence(const double tol = 1e-4, const unsigned maxiterations = 100)
209  : tolerance(tol),
210  maxiterations(maxiterations)
211  { reset(); }
212  virtual ~TestConvergence() {}
213 
214 public:
215  virtual bool operator()(const double new_stress, std::valarray<double> & X, std::valarray<double> & Y)
216  {
217  COLA_UNUSED(X);
218  COLA_UNUSED(Y);
219 
220  iterations++;
221  //std::cout<<"iteration="<<iterations<<", old_stress="<<old_stress<<", new_stress="<<new_stress<<std::endl;
222  if (old_stress == DBL_MAX) {
223  old_stress = new_stress;
224  return iterations >= maxiterations;
225  }
226  // converged if relative decrease in stress falls below threshold
227  // or if stress increases (shouldn't happen for straight majorization)
228  bool converged =
229  (old_stress - new_stress) / (new_stress + 1e-10) < tolerance
230  || iterations > maxiterations;
231  old_stress = new_stress;
232  return converged;
233  }
234  void reset() {
235  old_stress = DBL_MAX;
236  iterations = 0;
237  }
238  const double tolerance;
239  const unsigned maxiterations;
240  unsigned iterations;
241 };
